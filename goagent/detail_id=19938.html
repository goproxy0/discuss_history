# 安装依赖包
apt-get install geoip-bin sshpass

cd psiphon

# 创建 psi_sl.py
cat > psi_sl.py << EOF
#!/usr/bin/python

from psi_api import Psiphon3Server
from psi_ssh_connection import SSHConnection, OSSHConnection
import json
import os

DATA_FILENAME = 'psi_client.dat'
CLIENT_VERSION = 1
CLIENT_PLATFORM = 'Python'

class Data(object):

    def __init__(self, data):
        self.data = data

    @staticmethod
    def load():
        try:
            with open(DATA_FILENAME, 'r') as data_file:
                data = Data(json.loads(data_file.read()))
            # Validate
            data.servers()[0]
            data.propagation_channel_id()
            data.sponsor_id()
        except (IOError, ValueError, KeyError, IndexError, TypeError) as error:
            print '\nPlease obtain a valid %s file and try again.\n' % (DATA_FILENAME,)
            raise
        return data

    def save(self):
        with open(DATA_FILENAME+'.new', 'w') as data_file:
            data_file.write(json.dumps(self.data))
        os.rename(DATA_FILENAME+'.new', DATA_FILENAME)

    def servers(self):
        return self.data['servers']

    def propagation_channel_id(self):
        return self.data['propagation_channel_id']

    def sponsor_id(self):
        return self.data['sponsor_id']

    def move_first_server_entry_to_bottom(self):
        servers = self.servers()
        if len(servers) > 1:
            servers.append(servers.pop(0))
            return True
        else:
            return False


def print_server_list():

    data = Data.load()
    for _ in data.servers():
        server = Psiphon3Server(data.servers(), data.propagation_channel_id(), data.sponsor_id(), CLIENT_VERSION, CLIENT_PLATFORM)

        print 'sshpass -p %s ./psi_ssh -o StrictHostKeyChecking=no -o ConnectTimeout=10 -f -C -D localhost:7777 -N -p %s -z -Z %s %s@%s' % (server.get_password(), server.get_obfuscated_ssh_port(), server.get_obfuscate_keyword(), server.get_username(), server.get_ip_address())

        if not data.move_first_server_entry_to_bottom():
            break
        data.save()


if __name__ == "__main__":
    print_server_list()

EOF


# 创建 psi_test.sh
cat > psi_test.sh << EOF
#!/bin/bash

psi_speed()
{
        echo "("$(curl --socks5 localhost:7777 -s --connect-timeout 20 -Y 20000 -o /dev/null -w '%{speed_download}' https://www.youtube.com)")/1" | bc
}

ip_geo()
{
        GEO=$(geoiplookup $1 2>/dev/null | awk -F": " '/Country/ {print substr($2, 1, 2)}')

        if [ "$GEO" = IP ]; then
                return 1
        else
                echo $GEO
                return 0
        fi
}


rm -rf psl
rm -rf psl.speed

[ "$(pidof psi_ssh)" ] &&
        killall -KILL psi_ssh

python psi_sl.py > psl.orig

while read -r cmd; do
        echo $cmd >> psl
done < psl.orig

[ -e psl.sort ] && {
        while read -r speed country cmd; do
                echo $cmd >> psl
        done < psl.sort
}

cat psl | sort | uniq > psl.uniq
total=$(cat psl.uniq | wc -l)

while read -r cmd; do
        ((++count))

        $cmd &
        pid=$!
        cnt=0
        ret=0
        while :; do
                (kill -0 $pid) 2>/dev/null
                [ $? != 0 ] && break

                if ((cnt < 60)); then
                        sleep 1
                        ((cnt++))
                else
                        (kill -0 $pid) 2>/dev/null && {
                                (kill $pid) 2>/dev/null
                                sleep 1
                                (kill -0 $pid) 2>/dev/null && {
                                        (killall -KILL $pid) 2>/dev/null
                                        echo ossh: timeout
                                        ret=1
                                }
                        }
                fi
        done

        wait $pid
        if [ $? = 0 -a $ret = 0 ]; then
                speed=$(printf %09d $(psi_speed))
                country=$(ip_geo $(echo $cmd | awk -F@ '{print $2}'))
                echo test: $count/$total $speed $country $cmd
                echo $speed $country $cmd >> psl.speed
        fi

        [ "$(pidof psi_ssh)" ] && killall -KILL psi_ssh
done < psl.uniq

sort -r psl.speed > psl.sort
sed -e 's/psi_ssh/ssh/' -e 's/:7777/:1080/' psl.sort > psi_server_list.sort
EOF

chmod +x psi_test.sh

# 运行
./psi_test.sh
