#6 fafaf...@gmail.com 
临时解决问题

#!/usr/bin/env perl
use strict;
use warnings;

my $goagent = "python /Volumes/Home/goagent/goagent-32218/local/proxy.py";
my $pidFind = `ps -e -o pid,comm|grep python`;

if ($pidFind eq '') {

    system ($goagent);

}else{

    my ($pid, undef) = split(' ', $pidFind);
    my $cpuLoad = `ps -e -o pid,pcpu |grep $pid`;
    my (undef,$cpuUsage) = split(' ',$cpuLoad);

    if ($cpuUsage >= 92 ) {
        print "Killing pid $pid of )\n";
        system ("kill -9 $pid");
        system ($goagent);
    }
}
