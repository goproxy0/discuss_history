#4 wxj...@gmail.com 
有时是很难上传
可以把uploader.bat修改一下

@echo off

set uploaddir=python

( 
    echo ===============================================================
    echo  GoAgent服务端部署程序, 开始上传%uploaddir%服务端
    echo  如果需要上传golang服务端, 请修改本文件的uploaddir的值为golang
    echo ===============================================================
    echo.
    echo 请输入您的appid, 多个appid请用^|号隔开
) && (
    @cd /d "%~dp0" 
) && (
    set HTTP_PROXY=http://127.0.0.1:8087
    set HTTPS_PROXY=http://127.0.0.1:8087
) && (
    set PYTHONSCRIPT="import sys;sys.path.insert(0, 'uploader.zip');import appcfg;appcfg.main()"
) && (
    "..\local\proxy.exe"
) && (
    echo 上传成功，请编辑proxy.ini把你的appid填进去，谢谢。
)

@pause>NUL

  
  
@echo off



然后先运行goagent.exe,然后再运行uploader.bat,要顺利得多
