不明白新版为何放弃了通过goagent代理中转的方式
难道说是goagent的默认账户不够用了?既然中转的方式可用,那么不通过goagent中转,直接指定appengine.gooogle.com的IP应该也可以吧,不知如何实现,似乎改了系统hosts还是不行呀
于是我只能把uploader.bat改成这样了
@echo off

set uploaddir=golang

( 
    echo ===============================================================
    echo  GoAgent服务端部署程序, 开始上传%uploaddir%服务端
    echo  如果需要上传python服务端, 请修改本文件的uploaddir的值为python
    echo ===============================================================
    echo.
    echo 请输入您的appid, 多个appid请用^|号隔开
) && (
    @cd /d "%~dp0" 
) && (
    set HTTP_PROXY=http://127.0.0.1:8087
    set HTTPS_PROXY=http://127.0.0.1:8087
) && (
    set PYTHONSCRIPT=uploader.zip
) && (
    "..\local\proxy.exe"
)
|| (
    pause
)
  
  
@echo off
