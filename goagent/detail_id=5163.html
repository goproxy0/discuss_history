#2 twfe...@gmail.com 
我来告诉你几个有用而且可能会更改到的项目的含义

[listen] <-- 监听，表示监听本地的IP与端口 
ip = 127.0.0.1  <-- 本机LocalHost IP，你可以修改，但必须是本地IP
port = 8087  <-- Goagent 默认8087端口，你可以修改
visible=0  <-- 是否显示托盘图标

[gae]  <-- GAE配置
enable = 1  <-- 是否启用
appid = agence-01  <-- 你的Google engine AppID,也就是服务器部署的APPID
password =   <-- 密码,默认为空,你可以在上传的Server设定,如果设定了,此处需要与Server保持一致
path = /fetch.py  <-- 处理你请求的py文件,不用修改,如果不懂也不要修改.
profile = google_cn  <-- 启用哪个代理配置档
mulconn = 1 <-- 连接并发数
rangesize = 4194304  <-- 队列大小,如无必要,请默认

[paas]  <-- php配置
enable = 0 <-- 是否启用
password = 123456 <-- 密码,默认为空,你可以在上传的Server设定,如果设定了,此处需要与Server保持一致
listen = 127.0.0.1:8088 监听，表示监听本地的IP与端口 
fetchserver = https://app.com/ <-- php server的地址

[proxy]  <-- 二级代理,一般公司内网会用到
enable = 0    <-- 是否启用
host = 10.64.1.63  <-- 代理服务器地址
port = 8080   <-- 代理服务器端口
username = username  <-- 帐号 (注意,如果是NTLM,需要输入"网域\帐号"
password = 123456 <-- 密码

[google_cn]   <-- GAE配置档 ,与这个有关profile = google_cn 
mode = https  <-- 使用http / https (SSL加密)
hosts = 203.208.46.1|203.208.46.2|203.208.46.3|203.208.46.4|203.208.46.5|203.208.46.6|203.208.46.7|203.208.46.8   <-- google 服务器IP
sites = .google.com|.googleusercontent.com|.googleapis.com|.google-analytics.com|.googlecode.com|.google.com.hk|.appspot.com|.android.com|.googlegroups.com   
forcehttps = groups.google.com|code.google.com|mail.google.com|docs.google.com|profiles.google.com|developer.android.com  <--强制使用https访问
withgae = plus.google.com|plusone.google.com|reader.googleusercontent.com|music.google.com|apis.google.com  <-- 使用gae,显示为google本地

[google_hk] (同上)
mode = https
hosts = www.google.com|mail.google.com|www.google.com.hk|www.google.com.tw
sites = .google.com|.googleusercontent.com|.googleapis.com|.google-analytics.com|.googlecode.com|.google.com.hk|.googlegroups.com
forcehttps = www.google.com|groups.google.com|code.google.com|mail.google.com|docs.google.com|profiles.google.com|developer.android.com|www.google.com.hk|webcache.googleusercontent.com

#6 twfe...@gmail.com 
"那么说，GoAgent使用的Google DNS服务器，是内置的了，用户无法干预了，对吧？"
应该来说是不需要你干预.具体DNS解析使用的是哪台服务器,就看google所在的网络位置,为其提供服务器的ISP商了.

关于DNS解析,如果你了解一些基础网络的基础知识的话就知道,全球有13台ROOT DNS解析服务器(都不在中国大陆,也幸亏不在中国大陆),中国大陆使用的是镜像DNS解析服务器,相当于根域名解析服务器的子节点,所以为什么我们的DNS会被劫持或者污染,也是这些子节点放置在国内造成的.
