#7 jyuan...@gmail.com 
#5的方法可行，3.1.19需要修改三处，最好重新下载一个3.1.19来修改。

修改前：
1645     ssl_sock = ssl.wrap_socket(sock, ssl_version=self.ssl_version, do_handshake_on_connect=False)
1647     ssl_sock = ssl.wrap_socket(sock, ssl_version=self.ssl_version, cert_reqs=ssl.CERT_REQUIRED, ca_certs=os.path.join(os.path.dirname(os.path.abspath(__file__)), 'cacert.pem'), do_handshake_on_connect=False)

1817     thread.start_new_thread(create_connection_withopenssl, (addr, timeout, queobj))

修改后：
1645     ssl_sock = ssl.wrap_socket(sock, ssl_version=self.ssl_version, ciphers='ECDHE-RSA-AES128-SHA', do_handshake_on_connect=False)
1647     ssl_sock = ssl.wrap_socket(sock, ssl_version=self.ssl_version, ciphers='ECDHE-RSA-AES128-SHA', cert_reqs=ssl.CERT_REQUIRED, ca_certs=os.path.join(os.path.dirname(os.path.abspath(__file__)), 'cacert.pem'), do_handshake_on_connect=False)

1817     #thread.start_new_thread(create_connection_withopenssl, (addr, timeout, queobj))
1818     thread.start_new_thread(create_connection, (addr, timeout, queobj))

1818为新增加的。

我还是强烈建议你升级到3.2.x版本，因为新版对dns的处理相比旧版有很大不同，配合dns反污染软件能更好抵御共匪的dns投毒这种极其下贱的行为。当然你需要修改proxy.ini那里的dns设置为servers = 127.0.0.1，不要用goa自带的dns服务器，在我这里效果非常好。
